001package com.oreilly.config;
002
003import com.oreilly.repositories.AccountRepository;
004import org.apache.commons.dbcp2.BasicDataSource;
005import org.springframework.beans.factory.annotation.Autowired;
006import org.springframework.beans.factory.config.BeanPostProcessor;
007import org.springframework.context.annotation.Bean;
008import org.springframework.context.annotation.ComponentScan;
009import org.springframework.context.annotation.Configuration;
010import org.springframework.context.annotation.PropertySource;
011import org.springframework.core.env.Environment;
012import org.springframework.dao.annotation.PersistenceExceptionTranslationPostProcessor;
013import org.springframework.data.jpa.repository.config.EnableJpaRepositories;
014import org.springframework.jdbc.datasource.DataSourceTransactionManager;
015import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseBuilder;
016import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseType;
017import org.springframework.orm.jpa.JpaTransactionManager;
018import org.springframework.orm.jpa.JpaVendorAdapter;
019import org.springframework.orm.jpa.LocalContainerEntityManagerFactoryBean;
020import org.springframework.orm.jpa.vendor.Database;
021import org.springframework.orm.jpa.vendor.HibernateJpaVendorAdapter;
022import org.springframework.transaction.PlatformTransactionManager;
023import org.springframework.transaction.annotation.EnableTransactionManagement;
024
025import javax.persistence.EntityManagerFactory;
026import javax.sql.DataSource;
027import java.util.Properties;
028
029@Configuration
030@ComponentScan(basePackages = "com.oreilly")
031@EnableJpaRepositories(basePackages = "com.oreilly.repositories")
032@EnableTransactionManagement
033public class AppConfig {
034
035    @Bean
036    public DataSource dataSource() {
037        return new EmbeddedDatabaseBuilder()
038                .generateUniqueName(true)
039                .setType(EmbeddedDatabaseType.H2)
040                .setScriptEncoding("UTF-8")
041                .ignoreFailedDrops(true)
042                .addScript("schema.sql")
043                .addScripts("data.sql")
044                .build();
045    }
046
047    @Bean
048    public JpaVendorAdapter jpaVendorAdapter() {
049        HibernateJpaVendorAdapter adapter = new HibernateJpaVendorAdapter();
050        adapter.setShowSql(true);
051        adapter.setGenerateDdl(true);
052        adapter.setDatabase(Database.H2);
053        adapter.setDatabasePlatform("org.hibernate.dialect.H2Dialect");
054        return adapter;
055    }
056
057    @Bean
058    public LocalContainerEntityManagerFactoryBean entityManagerFactory(
059            DataSource dataSource, JpaVendorAdapter jpaVendorAdapter) {
060
061        Properties props = new Properties();
062        props.setProperty("hibernate.format_sql", String.valueOf(true));
063        
064        LocalContainerEntityManagerFactoryBean emf =
065                new LocalContainerEntityManagerFactoryBean();
066        emf.setDataSource(dataSource);
067        emf.setPackagesToScan("com.oreilly.entities");
068        emf.setJpaVendorAdapter(jpaVendorAdapter);
069        emf.setJpaProperties(props);
070
071        return emf;
072    }
073
074    @Bean
075    public PlatformTransactionManager transactionManager(EntityManagerFactory emf) {
076        return new JpaTransactionManager(emf);
077    }
078
079    @Bean
080    public BeanPostProcessor persistenceTranslation() {
081        return new PersistenceExceptionTranslationPostProcessor();
082    }
083
084}
