001package com.oreilly.config;
002
003import com.oreilly.entities.*;
004import org.springframework.beans.factory.annotation.Autowired;
005import org.springframework.beans.factory.annotation.Qualifier;
006import org.springframework.context.annotation.*;
007
008import javax.sql.DataSource;
009import java.text.NumberFormat;
010import java.util.List;
011import java.util.Set;
012
013@Configuration
014@ComponentScan(basePackages = "com.oreilly")
015@EnableAspectJAutoProxy
016public class AppConfig {
017    @Autowired
018    private DataSource dataSource;
019
020    @Autowired
021    private List<Team> teams;
022
023    @Bean
024    public NumberFormat nf() {
025        return NumberFormat.getCurrencyInstance();
026    }
027
028    @Bean
029    @Scope("prototype")
030    public Game game() {
031        BaseballGame baseballGame = new BaseballGame(teams.get(0),teams.get(1));
032        baseballGame.setDataSource(dataSource);
033        return baseballGame;
034    }
035}
