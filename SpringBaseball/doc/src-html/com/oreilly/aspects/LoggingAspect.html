041@Aspect
042@Component
043public class LoggingAspect {
044    private Logger logger = Logger.getLogger(getClass().getName());
045
046    @Before("execution(void com.oreilly..*.set*(*))")
047    public void callSetters(JoinPoint joinPoint) {
048        String method = joinPoint.getSignature().getName();
049        String arg = joinPoint.getArgs()[0].toString();
050        logger.info("Called " + method + " with arg " + arg +
051            " on " + joinPoint.getTarget());
052    }
053    /**
054     * Allows running aspects before and after bean execution.
055     * @param pjp Join Point for operation
056     * @return Result of operation
057     * @throws Throwable
058     */
059    @Around("execution(String playGame())")
060    public Object checkForRain(ProceedingJoinPoint pjp) throws Throwable {
061        boolean rain = Math.random() < 0.01;
062        Object result = null;
063        if (rain) {
064            logger.info(pjp.getTarget() + " rained out");
065        } else {
066            result = pjp.proceed();
067            logger.info(result.toString());
068        }
069        return result;
070    }
071}
