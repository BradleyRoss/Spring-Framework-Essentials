001package com.oreilly;
002
003import com.oreilly.entities.Game;
004import com.oreilly.entities.Team;
005import com.oreilly.entities.Cubs;
006import com.oreilly.entities.RedSox;
007import com.oreilly.entities.BaseballGame;
008/**
009 * Runs the code for the baseball games without
010 * using the Spring framework.
011 * 
012 * <p>This code directly uses the classes that are
013 *    used in the Spring framework to create the beans.</p>
014 * 
015 *
016 */
017public class RunDemoWithoutSpring {
018        public static void main(String[] args) {
019                Team redSox = new RedSox();
020                Team cubs = new Cubs();
021                Game game = new BaseballGame(redSox, cubs);
022                for (int i = 0; i < 4; i++) {
023                        System.out.println(game.playGame());
024                }
025        }
026}
