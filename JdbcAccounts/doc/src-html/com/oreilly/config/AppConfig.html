001package com.oreilly.config;
002

006import org.apache.commons.dbcp2.BasicDataSource;
007import org.springframework.beans.factory.annotation.Autowired;
008import org.springframework.context.annotation.*;
009import org.springframework.core.env.Environment;
010import org.springframework.jdbc.datasource.AbstractDataSource;
011import org.springframework.jdbc.datasource.DataSourceTransactionManager;
012import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseBuilder;
013import org.springframework.jdbc.datasource.embedded.EmbeddedDatabaseType;
014import org.springframework.transaction.PlatformTransactionManager;
015import org.springframework.transaction.annotation.EnableTransactionManagement;
016import java.util.Properties;
017
018import javax.sql.DataSource;
019/**
020 * Instantiates Spring beans.
021 * 
022 * @see Autowired
023 * @see Bean
024 * @see ComponentScan
025 * @see Configuration
026 * @see Environment
027 * @see PropertySource
028 * @see EnableTransactionManagement
029 *
030 */
031@Configuration
032@ComponentScan(basePackages = "com.oreilly")
033@PropertySource("classpath:prod.properties")
034@EnableTransactionManagement
035public class AppConfig {
036        /**
037         * Bean reading the properties from a {@link Properties} file in
038         * the classpath.
039         */
040    @Autowired
041    private Environment env;
042    /**
043     * Configures bean for test data source.
044     * @return Bean
045     * @see EmbeddedDatabaseBuilder
046     */
047    @Bean(name = "dataSource", destroyMethod = "shutdown")
048    @Profile("test")
049    public DataSource dataSourceForTest() {
050        return new EmbeddedDatabaseBuilder()
051                .generateUniqueName(true)
052                .setType(EmbeddedDatabaseType.H2)
053                .setScriptEncoding("UTF-8")
054                .ignoreFailedDrops(true)
055                .addScript("schema.sql")
056                .addScripts("data.sql")
057                .build();
058    }
059    /**
060     * Configures bean for test transaction manager.
061     * @return bean
062     */
063    @Bean(name = "transactionManager")
064    @Profile("test")
065    public PlatformTransactionManager transactionManagerForTest() {
066        return new DataSourceTransactionManager(dataSourceForTest());
067    }
068    /**
069     * Configures bean for production data source.
070     * @return bean
071     */
072    @Bean(name = "dataSource")
073    @Profile("prod")
074    public BasicDataSource dataSourceForProd() {
075        BasicDataSource dataSource = new BasicDataSource();
076        dataSource.setDriverClassName(env.getProperty("db.driver"));
077        dataSource.setUrl(env.getProperty("db.url"));
078        dataSource.setUsername(env.getProperty("db.user"));
079        dataSource.setPassword(env.getProperty("db.pass"));
080        return dataSource;
081    }
082    /**
083     * Configures bean for production transaction manager.
084     * @return bean
085     */
086    @Bean(name = "transactionManager")
087    @Profile("prod")
088    public PlatformTransactionManager transactionManagerForProd() {
089        return new DataSourceTransactionManager(dataSourceForProd());
090    }
091
092}
