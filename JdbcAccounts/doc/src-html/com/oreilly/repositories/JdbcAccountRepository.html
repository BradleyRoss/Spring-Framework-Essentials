001package com.oreilly.repositories;
002
003import com.oreilly.entities.Account;
004import org.springframework.beans.factory.annotation.Autowired;
005import org.springframework.context.annotation.Profile;
006import org.springframework.jdbc.core.JdbcTemplate;
007import org.springframework.jdbc.core.RowMapper;
008import org.springframework.stereotype.Repository;
009
010import javax.sql.DataSource;
011import java.math.BigDecimal;
012import java.sql.ResultSet;
013import java.sql.SQLException;
014import java.util.List;
015/**
016 * This is the code for the repository bean using the 
017 * prod and test profiles.
018 * 
019 * 
020 *
021 */
022@Repository
023@Profile({"prod", "test"})
024public class JdbcAccountRepository implements AccountRepository {
025    private JdbcTemplate template;
026    private static long nextId = 4;
027    /*
028     * Can a constructor be Autowired?
029     */
030   //  @Autowired
031    public JdbcAccountRepository(DataSource dataSource) {
032        template = new JdbcTemplate(dataSource);
033    }
034
035    @Override
036    public List<Account> getAccounts() {
037        String sqlTxt = "select * from account";
038        return template.query(sqlTxt, new AccountMapper());
039    }
040
041    @Override
042    public Account getAccount(Long id) {
043        String sqlTxt = "select * from account where id=?";
044        return template.queryForObject(sqlTxt, new AccountMapper(), id);
045    }
046
047
048    @Override
049    public int getNumberOfAccounts() {
050        String sqlTxt = "select count(*) from account";
051        return template.queryForObject(sqlTxt, Integer.class);
052    }
053    /**
054     * Method for creating account.
055     * 
056     * <p>Since the value returned by template.update is not
057     *    used it is not necessary to place the return value
058     *    in a variable.</p>
059     */
060    @Override
061    public Long createAccount(BigDecimal initialBalance) {
062        String sqlTxt = "insert into account(id,balance) values(?,?)";
063        long id = nextId++;
064        // int uc = template.update(sqlTxt, id, initialBalance);
065        template.update(sqlTxt, id, initialBalance);
066        return id;
067    }
068
069    @Override
070    public void updateAccount(Account account) {
071        String sqlTxt = "update account set balance = ? where id = ?";
072        template.update(sqlTxt, account.getBalance(), account.getId());
073    }
074
075    @Override
076    public int deleteAccount(Long id) {
077        String sqlTxt = "delete from account where id=?";
078        return template.update(sqlTxt, id);
079    }
080
081    private class AccountMapper implements RowMapper<Account> {
082        @Override
083        public Account mapRow(ResultSet resultSet, int i) throws SQLException {
084            return new Account(resultSet.getLong("id"),
085                    resultSet.getBigDecimal("balance"));
086        }
087    }
088}
