001package com.oreilly.repositories;
002
003import com.oreilly.entities.Account;
004import org.springframework.context.annotation.Profile;
005import org.springframework.stereotype.Repository;
006
007import java.math.BigDecimal;
008import java.util.ArrayList;
009import java.util.Collection;
010import java.util.List;
011import java.util.Map;
012import java.util.concurrent.ConcurrentHashMap;
013/**
014 * This implementation of {@link AccountRepository} interface
015 * is used with the
016 * development profile.
017 * 
018 * It stores the contents of the repository as Java data structures.
019 * 
020 *
021 */
022@Repository
023@Profile("dev")
024public class InMemoryAccountRepository implements AccountRepository {
025        /**
026         * Storage of account data.
027         */
028    private static Map<Long, Account> accountMap =
029            new ConcurrentHashMap<Long, Account>();
030    /**
031     * Next account number to be issued.
032     */
033    private static Long nextId = 4L;
034    /*
035     * Initializes account information.
036     */
037    static {
038        accountMap.put(1L, new Account(1L, new BigDecimal("100.0")));
039        accountMap.put(2L, new Account(2L, new BigDecimal("120.0")));
040        accountMap.put(3L, new Account(3L, new BigDecimal("150.0")));
041    }
042
043    @Override
044    public List<Account> getAccounts() {
045        Collection<Account> accountCollection = accountMap.values();
046        List<Account> accountList = new ArrayList<Account>();
047        for (Account account : accountCollection) {
048            accountList.add(account);
049        }
050        return accountList;
051    }
052
053    @Override
054    public Account getAccount(Long id) {
055        return accountMap.get(id);
056    }
057
058    @Override
059    public int getNumberOfAccounts() {
060        return accountMap.size();
061    }
062
063    @Override
064    public Long createAccount(BigDecimal initialBalance) {
065        Long id = nextId++;
066        accountMap.put(id, new Account(id, initialBalance));
067        return id;
068    }
069
070    @Override
071    public int deleteAccount(Long id) {
072        accountMap.remove(id);
073        return 1;
074    }
075
076    @Override
077    public void updateAccount(Account account) {
078        accountMap.put(account.getId(), account);
079    }
080}
