001package com.oreilly.services;
002
003import com.oreilly.entities.Account;
004import com.oreilly.repositories.AccountRepository;
005import org.springframework.beans.factory.annotation.Autowired;
006import org.springframework.context.annotation.Profile;
007import org.springframework.stereotype.Service;
008import org.springframework.stereotype.Repository;
009import org.springframework.transaction.annotation.Propagation;
010import org.springframework.transaction.annotation.Transactional;
011
012import java.math.BigDecimal;
013/**
014 * The methods of this class are called by the
015 * business logic.  
016 * 
017 * <p>The methods in the
018 *    bean identified by the {@link Service} annotation
019 *    then call the
020 *    appropriate methods in the bean
021 *    identified by the {@link Repository}.</p>
022 * 
023 *
024 */
025@Service @Transactional
026@Profile("test")
027public class AccountService {
028
029    @Autowired
030    private AccountRepository repository;
031
032    @Transactional(readOnly = true, propagation = Propagation.SUPPORTS)
033    public BigDecimal getBalance(Long id) {
034        return repository.getAccount(id).getBalance();
035    }
036
037    public BigDecimal deposit(Long id, BigDecimal amount) {
038        Account account = repository.getAccount(id);
039        BigDecimal newBalance = account.getBalance().add(amount);
040        account.setBalance(newBalance);
041        repository.updateAccount(account);
042        return newBalance;
043    }
044
045    public BigDecimal withdraw(Long id, BigDecimal amount) {
046        return deposit(id, amount.negate());
047    }
048
049    public void transfer(Long fromId, Long toId, BigDecimal amount) {
050        withdraw(fromId, amount);
051        deposit(toId, amount);
052    }
053}
