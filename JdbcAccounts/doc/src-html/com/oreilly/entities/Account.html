001package com.oreilly.entities;
002
003import java.math.BigDecimal;
004
005public class Account {
006    private Long id;
007    private BigDecimal balance;
008    
009    public Account() {}
010
011    public Account(Long id, BigDecimal balance) {
012        this.id = id;
013        this.balance = balance;
014    }
015
016    public BigDecimal getBalance() {
017        return balance;
018    }
019
020    public void setBalance(BigDecimal balance) {
021        this.balance = balance;
022    }
023    
024    public Long getId() {
025        return id;
026    }
027
028    @Override
029    public String toString() {
030        return String.format("Account(id:%d, balance:%s)", id, balance);
031    }
032
033    @Override
034    public boolean equals(Object o) {
035        if (this == o) return true;
036        if (o == null || getClass() != o.getClass()) return false;
037
038        Account account = (Account) o;
039
040        return !(id != null ? !id.equals(account.id) : account.id != null);
041
042    }
043
044    @Override
045    public int hashCode() {
046        return id != null ? id.hashCode() : 0;
047    }
048}
