001package com.oreilly;
002
003import com.oreilly.config.AppConfig;
004import org.springframework.context.ApplicationContext;
005import org.springframework.context.annotation.AnnotationConfigApplicationContext;
006// import org.springframework.context.annotation.Profile;
007import org.springframework.context.support.AbstractApplicationContext;
008// import org.springframework.jdbc.datasource.AbstractDataSource;
009import javax.sql.DataSource;
010import java.util.List;
011import java.util.ArrayList;
012/**
013 * Test driver.
014 * 
015 * <p>Changes</p>
016 * <ul>
017 * <li>Changed {@link ApplicationContext} to {@link AbstractApplicationContext}</li>
018 * <li>Added import statement for AbstractApplicationContext</li>
019 * <li>Closed context at end of program</li>
020 * </ul>
021 * @see AnnotationConfigApplicationContext
022 *
023 */
024public class RunDemo implements Runnable {
025        protected List<String> profileList = new ArrayList<String>();
026        protected void setProfiles(List<String> values) {
027                profileList = values;
028        }
029        protected List<String> getProfiles() {
030                return profileList;
031        }
032        public void run() {
033                if (profileList.size() == 0) {
034                        profileList.add("prod");
035                }
036                for (String item : profileList) {
037                        runCase(item);
038                        for (int i = 0; i < 2; i++) {
039                                System.out.println("*** *** *** *** *** *** *** ***");
040                        }
041                }
042        }
043        protected void runCase(String profile) {
044                System.setProperty("spring.profiles.active", profile);
045                System.out.println("Running profile " + profile);
046                AbstractApplicationContext context =
047                                new AnnotationConfigApplicationContext(AppConfig.class);
048
049                System.out.println("There are " + 
050                                context.getBeanDefinitionCount() + " beans");
051                for (String name : context.getBeanDefinitionNames()) {
052                        System.out.println("     " + name);
053                }
054                System.out.println("Looking for beans implementing DataSource");
055                String[] beanList = context.getBeanNamesForType(DataSource.class);
056                if (beanList.length == 0) {
057                        System.out.println("     No beans found");
058                } else {
059                        for (String item : beanList) {
060                                System.out.println("     " + item);
061                        }
062                }
063                context.close();
064        }
065        public static void main(String[] args) {
066                List<String> profiles = new ArrayList<String>();
067                profiles.add("prod");
068                profiles.add("test");
069                profiles.add("dev");
070                RunDemo instance = new RunDemo();
071                instance.setProfiles(profiles);
072                instance.run();
073        }
074}
