001package com.oreilly.services;
002
003import com.oreilly.entities.Account;
004import com.oreilly.repositories.AccountRepository;
005import org.springframework.beans.factory.annotation.Autowired;
006import org.springframework.context.annotation.Profile;
007import org.springframework.stereotype.Service;
008import org.springframework.transaction.annotation.Propagation;
009import org.springframework.transaction.annotation.Transactional;
010
011import java.math.BigDecimal;
012
013@Service @Transactional
014public class AccountService {
015
016    @Autowired
017    private AccountRepository repository;
018
019    @Transactional(readOnly = true, propagation = Propagation.SUPPORTS)
020    public BigDecimal getBalance(Long id) {
021        return repository.getAccount(id).getBalance();
022    }
023
024    public BigDecimal deposit(Long id, BigDecimal amount) {
025        Account account = repository.getAccount(id);
026        BigDecimal newBalance = account.getBalance().add(amount);
027        account.setBalance(newBalance);
028        repository.updateAccount(account);
029        return newBalance;
030    }
031
032    public BigDecimal withdraw(Long id, BigDecimal amount) {
033        return deposit(id, amount.negate());
034    }
035
036    public void transfer(Long fromId, Long toId, BigDecimal amount) {
037        withdraw(fromId, amount);
038        deposit(toId, amount);
039    }
040}
